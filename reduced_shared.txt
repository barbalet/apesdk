#ifndef SHARED_H
#define SHARED_H

#include <stdint.h>

// View type constants
#define NUM_VIEW        0
#define NUM_TERRAIN     1
#define NUM_CONTROL     2

// Menu constants
#define NA_MENU_PAUSE           100
#define NA_MENU_PREVIOUS_APE    101
#define NA_MENU_NEXT_APE        102
#define NA_MENU_CLEAR_ERRORS    103
#define NA_MENU_TERRITORY       104
#define NA_MENU_WEATHER         105
#define NA_MENU_BRAINCODE       106

// Type definitions
typedef int32_t n_int;

// Core simulation functions
void shared_init(n_int view_type, unsigned int seed);
int shared_simulation_started(void);
int shared_cycle(unsigned int time_info, n_int view_type);

// Drawing functions
void* shared_draw(n_int view_type, int width, int height, int unused);

// Input handling functions
void shared_keyReceived(int key, n_int view_type);
void shared_keyUp(void);
void shared_mouseReceived(double x, double y, n_int view_type);
void shared_mouseOption(int option);
void shared_mouseUp(void);
void shared_delta(double delta_x, double delta_y, n_int view_type);
void shared_zoom(double magnification, n_int view_type);
void shared_rotate(double rotation, n_int view_type);

// Menu functions
int shared_menu(int menu_id);

// File operations
int shared_openFileName(const char* filename, int script_mode);
void shared_saveFileName(const char* filename);

// Simulation control
void shared_new(unsigned int seed);
void shared_about(void);

// Command line interface
void io_command_line_execution_set(void);

#endif // SHARED_H
